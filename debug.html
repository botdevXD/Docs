debug.getupvalues

table > debug.getupvalues(function b)

Returns a table with all the upvalues from b(function)

local a, b, c, d, e, f, g = 10, 20, 30, 40, 50, 60, 70
function Okay()
	print(a, b, c, d, e, f, g)
end

for i,v in pairs(debug.getupvalues(Okay)) do
	print(i, v)
end
-- 1 (index) 10 (upvalue)
-- 2 (index) 20 (upvalue)
-- 3 (index) 30 (upvalue)
-- 4 (index) 40 (upvalue)
-- 5 (index) 50 (upvalue)
-- 6 (index) 60 (upvalue)
-- 7 (index) 70 (upvalue)

debug.getupvalue

int > debug.getupvalue(function b, int index)

Returns an upvalue using the provided index(int)

local a, b, c, d, e, f, g = 10, 20, 30, 40, 50, 60, 70
function Okay()
    print(a, b, c, d, e, f, g)
end

print(debug.getupvalue(Okay, 1)) -- 10 (upvalue)

debug.getconstants

table > debug.getconstants(function b)

Fetches all the constants from b(function) & returns them.

function Okay()
    print("Coco Z3 winning?")
end

for i,v in pairs(debug.getconstants(Okay)) do
    if i == 1 then
        print(i, v) -- 1 print
    elseif i == 2 then
        print(i, v) -- 2 Coco Z3 winning?
    end
end -- you don't need to do the if i == 1 stuff.

debug.getconstant

void > debug.getconstant(function b, int index)

Fetches the constant from b(function) using index(int) & returns the constant.

function Okay()
    print("Coco Z3 winning?")
end

print(debug.getconstant(Okay, 2)) -- Coco Z3 winning?

debug.getprotos

table > debug.getprotos(function b)

Fetches the protos from b(function) & returns them.

function Okay()
    local function Coco()
        print("Coco Z3 winning?")
    end
end

for i,v in pairs(debug.getprotos(Okay)) do
    print(i,v)
end
-- 1 0xFunctionAddress

debug.setupvalue

void > debug.setupvalue(function b, int index, int new_value)

Sets an upvalue at the provided index(int), the upvalue will be set to new_value(int)

local a, b, c, d, e, f, g = 10, 20, 30, 40, 50, 60, 70
function Okay()
    print(a, b, c, d, e, f, g)
end

debug.setupvalue(Okay, 1, 50)
Okay() -- 50, 20, 30, 40, 50, 60, 70 (upvalues)

debug.setconstant

void > debug.setconstant(function b, int index, string new_constant)

Sets a constant using index(int), replaces old constant with new_constant(string)

function Okay()
    print("Coco Z3 winning?")
end

debug.setconstant(Okay, 2, "Coco Z3 is winning") -- Coco Z3 is winning
Okay()
